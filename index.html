Motion is a powerful, lightweight animation library that delivers smooth, high-performance animations through a minimal and intuitive API.

Features:

  • Simple API
  • Animations based on keyframes or physics
  • Animation sequences
  • Performance and accessibility optimized

// Base function structure
animate(target, keyframes, options);
Parameter / Option Type Description Example
target Element | string The element or CSS selector to animate. animate(".box", { x: 100 })
keyframes Object | Array Defines which properties to animate and their values (can be arrays for sequences). { opacity: [0, 1], x: [0, 100] }
options Object Animation settings like duration, easing, delay, repeat, etc. { duration: 0.8, ease: "easeOut" }
type string Defines the animation style. Can be "tween" (time-based) for smooth, consistent motion, or "spring" (physics-based) for natural, bouncy movement. { type: "spring", stiffness: 200, damping: 20 }
duration number Time in seconds for the animation. { duration: 1.2 }
delay number Delay before animation starts (in seconds). { delay: 0.3 }
repeat number | Infinity How many times to repeat the animation. { repeat: Infinity }
direction "normal" | "reverse" | "alternate" Direction of playback for repeating animations. { direction: "alternate" }
ease string | function Easing function controlling speed curve (e.g. spring, linear, easeInOut). { ease: "spring" }
stiffness number Controls how tight a spring animation is (higher = faster stop). { stiffness: 200 }
damping number How much resistance the spring has. { damping: 15 }
mass number Controls the weight of the spring animation (affects bounce feel). { mass: 1.5 }
bounce number Controls amount of bounce (0 to 1). Used with "spring" type. { bounce: 0.5 }
onUpdate function Called every frame with the latest animated value. onUpdate: (v) => console.log(v)
onComplete function Triggered once the animation finishes. onComplete: () => console.log("Done!")
{type: 'tween'} property
  • duration, ease, delay, repeat, direction, repeatDelay
 animate(".box", { y: [0, 100] }, { type: "tween", duration: 0.6, ease: "easeOut" })
{type: 'spring'} property
  • stiffness, damping, mass, bounce, velocity
 animate(".circle", { scale: [0, 1] }, { type: "spring", stiffness: 250, damping: 20 }) 

// On click reduce the scale of the button
const button1 = document.querySelector('#section1 .example-btn');

button1.addEventListener('click', function() {
    animate(this, { scale: 0.6 }, {
        type: 'spring',
        stiffness: 350,
        damping: 15,
    });
});
        

// On hover increase button size
const button2 = document.querySelector('#section2 .example-btn');

button2.addEventListener('mouseenter', function() {
    animate(this, { scale: 1.5  }, {
        duration: 0.3,
        easing: 'ease-out',
    });
});

// On mouse leave restore button size
button2.addEventListener('mouseleave', function() {
    animate(this, { scale: 1  }, {
        type: 'spring',
        bounce: '0.5'
    });
});
        

// On mouse down reduce button size and infinitely rotate through colors
const button3 = document.querySelector('#section3 .example-btn');
let colorAnimation = null;
let baseColor = '#00e1ff';
button3.addEventListener('mousedown', function () {
    // Scale down
    animate(this, { scale: 0.9 }, { duration: 0.2 });

    // Start looping color animation
    colorAnimation3 = animate(this, {
        backgroundColor: [baseColor3,'#FF6B6B', '#6a65a8ff', '#88dfa1ff', baseColor3]
    }, {
        duration: 2,
        easing: 'ease-out',
        delay: 1.5,
        repeat: Infinity
    });
});

// Restore the size and stop the color animation
button3.addEventListener('mouseup', function () {
    if (colorAnimation3) {
        colorAnimation3.stop();
        colorAnimation3 = null;
    }

    animate(this, { scale: 1, backgroundColor: baseColor3 }, { duration: 0.2 });
});
        

// Run 3 animations at the same time
const button4 = document.querySelector('#section4 .example-btn');

});

    ['#section7 .box', { x: [0, 150], rotate: [0, 180], backgroundColor: ['#8df0cc', '#d26565ff'] }, { type: "spring", stiffness: 250, damping: 15 }],

// Fade in/out when in/out of view
const { animate, inView } = Motion;

inView('#section8 .box', (element) => {
    animate(element, { scale: [0.5, 1], opacity: [0, 1], y: [-25, 0]}, { type: "spring", stiffness: 200, damping: 45, delay: 0.1 })

    return () =>
        animate(element, { scale: [1, 0.5], opacity: [1, 0], y: [-25, 0]}, { type: "spring", stiffness: 200, damping: 45 })
})
